#include <stdio.h>
#include <stdlib.h>
#include <math.h>

void eratos(int number, char array[])
{
	int n, m, to, last, index;

	to = (int)sqrt((double)number);
	for (n = 2; n <= to; n++)
	{
		last = number / n;
		for (m = 2; m <=last; m++)
		{
			index = n * m;
			array[index] = 1; 
		}
	}
}

void printp(int value, char* array)
{
	int n;
	int count = 0;

	printf("%d 이내의 소수 \n", value);

	for (n = 2; n <= value; n++)
	{
		if (array[n] == 0)
		{
			printf("%9d", n);
			count += 1;
			if ((count % 8) == 0)
				printf("\n");
		}
	}
	printf("\n");
}


void main()
{
	int number;
	char* array;

	printf("정수 입력 ");
	scanf_s("%d", &number);

	if (number >= 2)
	{
		array = (char*)calloc(number + 1, sizeof(char));
		if (array != NULL)
		{
			eratos(number, array);
			printp(number, array);
		}
	}
}
